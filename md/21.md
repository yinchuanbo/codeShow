```html
<div class="box">
  <div class="list">
    <div>1</div>
    <div>2</div>
    <div>3</div>
    <div>4</div>
    <div>5</div>
    <div>6</div>
  </div>
</div>
```

```css
#iframe__wrapper {
  display: flex;
  align-items: center;
  justify-content: center;
}

.box {
  margin: 20px auto;
  width: 300px;
  height: 80px;
  overflow: hidden;
  position: relative;
  border: 1px solid gray;
}

.list {
  position: absolute;
  left: 0;
  display: flex;
  align-items: center;
}

.list > div {
  width: 80px;
  height: 80px;
  border: 1px solid gray;
  box-sizing: border-box;
  color: #fff;
  text-align: center;
  line-height: 80px;
}
```

```js
let list = document.querySelector(".list");
let box = document.querySelector(".box");
list.innerHTML += list.innerHTML;
let left = 0;

let timer;

const move = () => {
  timer = setInterval(() => {
    left -= 2;
    if (left == -(6 * 80)) {
      left = 0;
    }
    list.style.left = left + "px";
  }, 20);
};

move();

box.onmouseenter = () => {
  clearInterval(timer);
};

box.onmouseleave = () => {
  move();
};
```
